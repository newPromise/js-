实现大小写转换主要用到的是字符串的 toUppercase() toLowerCase() 方法，下面的函数用于将字符串中的大写字母转化为小写，小写字母转化为大写
```
         function toggleCode(value) {
           var arr1='';
           var reg1 = /^[A-Z]$/;
           var reg2 = /^[a-z]$/;
           for (let i of  value) {
               if (reg1.test(i)) {
                   i = i.toLowerCase();
               } else  if (reg2.test(i)) {
                   i = i.toUpperCase();
               }
               arr1 = arr1 + i;
           }
           return arr1;
       }
       toggleCode('ZhangNing')// zHANGnING
             
```

2.使用for of 和 for in 的区别
观察下面的例子：
```
function Forloop(val) {
  for (let i of val) {
    console.log(i)
  }
  for (let i in val) {
    console.log(i)
  }
}
下面是三种情况：
1.数组
let  arr = [4,7,2];
Forloop(arr);
// 4,7,2   0,1,2
2.字符串
let  str = 'hello';
Forloop(str);
// 'h' 'e' 'l' 'l' 'o' 
// 0 1 2 3 4
3.对象
这个比较有意思，揭示了使用 for of 循环的原因
let obj = {};
Forloop(obj);
```
